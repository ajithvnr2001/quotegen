  user_id TEXT NOT NULL,
  action TEXT NOT NULL,

CREATE INDEX idx_usage_logs_timestamp ON usage_logs(timestamp);

function useDebounce<T>(value: T, delay: number): T {
  const [debouncedValue, setDebouncedValue] = useState<T>(value);

  useEffect(() => {
    const handler = setTimeout(() => {
      setDebouncedValue(value);
    }, delay);

    return () => {
      clearTimeout(handler);
    };
  }, [value, delay]);

  return debouncedValue;

    canvas.height = canvasHeight;

    // Clear and set background
    ctx.clearRect(0, 0, canvasWidth, canvasHeight);
    ctx.fillStyle = '#ffffff';
    ctx.fillRect(0, 0, canvasWidth, canvasHeight);

    // Calculate scale for cropping
    const scaleX = img.naturalWidth / img.width;
    const scaleY = img.naturalHeight / img.height;

    // Draw cropped image
    ctx.drawImage(
      img,
      completedCrop.x * scaleX,
      completedCrop.y * scaleY,
      completedCrop.width * scaleX,
      completedCrop.height * scaleY,
      0,
      0,
      canvasWidth,
      canvasHeight
    );

    // Enhanced Text Drawing
    const currentQuote = useCustomQuote ? customQuote : selectedQuote;
    if (currentQuote) {
      const scaledFontSize = (debouncedFontSize * canvasWidth) / 800;
      
      ctx.font = `${isItalic ? 'italic ' : ''}${fontWeight} ${scaledFontSize}px ${selectedFont}`;
      ctx.fillStyle = fontColor;
      ctx.textAlign = 'center';
      ctx.textBaseline = 'middle';

      // Apply text transformations
      ctx.save();
      ctx.translate(canvasWidth / 2 + (textPosition.x * canvasWidth / 800), canvasHeight / 2 + (textPosition.y * canvasHeight / 600));
      ctx.rotate((debouncedTextRotation * Math.PI) / 180);
      ctx.scale(mirrorText ? -1 : 1, 1);

      // Text wrapping
      const words = currentQuote.split(' ');
      const lines: string[] = [];
      let currentLine = '';
      const maxWidth = canvasWidth - 80;

      words.forEach(word => {
        const testLine = currentLine ? `${currentLine} ${word}` : word;
        if (ctx.measureText(testLine).width > maxWidth && currentLine) {
          lines.push(currentLine);
          currentLine = word;
        } else {
          currentLine = testLine;
        }
      });
      lines.push(currentLine);

      // Draw text lines
      const lineHeight = scaledFontSize * 1.2;
      const totalTextHeight = lines.length * lineHeight;
      const startY = -(totalTextHeight / 2) + (lineHeight / 2);

      lines.forEach((line, index) => {
        ctx.fillText(line, 0, startY + (index * lineHeight));
      });

      ctx.restore();
    }
  }, [
    completedCrop, selectedQuote, customQuote, useCustomQuote, 
    debouncedFontSize, fontColor, selectedFont, fontWeight, isItalic,
    textPosition, debouncedTextRotation, mirrorText, selectedSocialSize,
    customWidth, customHeight
  ]);

  // Generate Final Image

    if (!image) {
      alert('Please upload an image first');
      return;
    }

      return;
    }

    setLoading(true);

    try {
      const outputSize = getCurrentOutputSize();
      
      const response = await fetch(`${API_BASE}/api/generate`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          imageBase64: image,
          quoteText: finalQuote,
          fontId: selectedFont,
          fontSize: fontSize,
          fontColor: fontColor,
          fontWeight: fontWeight,
          isItalic: isItalic,
          userId,
          imageManipulation: {
            rotation: 0,
            scale: 1,
            cropX: completedCrop?.x || 0,
            cropY: completedCrop?.y || 0,
            cropWidth: completedCrop?.width || 0,
            cropHeight: completedCrop?.height || 0
          },
          textManipulation: {
            position: textPosition,
            rotation: textRotation,
            mirror: mirrorText
          },
          outputSize: {
            width: outputSize.width,
            height: outputSize.height,
            format: selectedSocialSize
          }
        })
      });

      if (response.ok) {
        const blob = await response.blob();
        const url = URL.createObjectURL(blob);
        setGeneratedImage(url);
      } else {

        throw new Error(`Generation failed: ${errorText}`);
      }
    } catch (error) {

      alert('Failed to generate image. Please try again.');

    }
